START
    VAR A INT ,B BOOLEAN,C DAY;
    A=2;
    B=TRUE;
    C=LUNI;
    READ A;
    WRITE TRUE;
    IF true or ((A or B) and false) == (1 + (3 + a))
    THEN
        B=FALSE;
        C=MARTI;
    ELSE
        A=TRUE;
    END-IF;
    WHILE true or ((A or B) and false) == (1 + (3 + a))
        THEN
            B=FALSE;
            C=MARTI;
    END-WHILE;
END
